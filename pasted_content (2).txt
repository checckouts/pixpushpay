        let currentPage = 1;
        const reviewsPerPage = 4;

        function renderReviews(page) {
            const grid = document.getElementById('reviewsGrid');
            if(!grid) return;
            grid.innerHTML = '';
            const startIndex = (page - 1) * reviewsPerPage;
            const pageReviews = allReviews.slice(startIndex, startIndex + reviewsPerPage);

            pageReviews.forEach(review => {
                const card = document.createElement('div');
                card.className = 'review-card';
                let starsHtml = '';
                for(let i=0; i<5; i++) starsHtml += `<i class="${i < review.rating ? 'fas' : 'far'} fa-star"></i>`;
                card.innerHTML = `<div class="review-top"><div class="reviewer-info"><div class="reviewer-avatar">${review.name.charAt(0)}</div><div><div class="reviewer-name">${review.name}</div><div class="verified-badge"><i class="fas fa-check-circle"></i> Verificada</div></div></div><div class="review-date">${review.date}</div></div><div class="review-stars" style="margin-bottom: 8px;">${starsHtml}</div><div class="review-content">${review.content}</div>`;
                grid.appendChild(card);
            });
            renderPagination(page);
        }

        function renderPagination(activePage) {
            const pagination = document.getElementById('pagination');
            if(!pagination) return;
            pagination.innerHTML = '';
            const totalPages = Math.ceil(allReviews.length / reviewsPerPage);
            
            let startPage = Math.max(1, activePage - 2);
            let endPage = Math.min(totalPages, startPage + 4);
            if (endPage - startPage < 4) startPage = Math.max(1, endPage - 4);

            if (startPage > 1) {
                const firstBtn = document.createElement('button');
                firstBtn.className = 'page-btn';
                firstBtn.textContent = '1';
                firstBtn.onclick = () => { currentPage = 1; renderReviews(1); window.scrollTo({top: document.getElementById('reviewsGrid').offsetTop - 150, behavior: 'smooth'}); };
                pagination.appendChild(firstBtn);
                if (startPage > 2) {
                    const dots = document.createElement('span');
                    dots.textContent = '...';
                    dots.style.alignSelf = 'center';
                    pagination.appendChild(dots);
                }
            }
            for (let i = startPage; i <= endPage; i++) {
                const btn = document.createElement('button');
                btn.className = `page-btn ${activePage === i ? 'active' : ''}`;
                btn.textContent = i;
                btn.onclick = () => { 
                    currentPage = i; 
                    renderReviews(i);
                    window.scrollTo({top: document.getElementById('reviewsGrid').offsetTop - 150, behavior: 'smooth'});
                };
                pagination.appendChild(btn);
            }

            if (endPage < totalPages) {
                if (endPage < totalPages - 1) {
                    const dots = document.createElement('span');
                    dots.textContent = '...';
                    dots.style.alignSelf = 'center';
                    pagination.appendChild(dots);
                }
                const lastBtn = document.createElement('button');
                lastBtn.className = 'page-btn';
                lastBtn.textContent = 'Ver mais';
                lastBtn.onclick = () => { currentPage = totalPages; renderReviews(totalPages); window.scrollTo({top: document.getElementById('reviewsGrid').offsetTop - 150, behavior: 'smooth'}); };
                pagination.appendChild(lastBtn);
            }
        }

        function openTab(evt, tabName) {
            const tabcontent = document.getElementsByClassName("tab-content");
            for (let i = 0; i < tabcontent.length; i++) { tabcontent[i].style.display = "none"; tabcontent[i].classList.remove("active"); }
            const tablinks = document.getElementsByClassName("tab-btn");
            for (let i = 0; i < tablinks.length; i++) tablinks[i].classList.remove("active");
            document.getElementById(tabName).style.display = "block";
            document.getElementById(tabName).classList.add("active");
            evt.currentTarget.classList.add("active");

            function toggleCart() {
                const isActive = cartSidebar.classList.contains('active');
                if (isActive) {
                    cartSidebar.classList.remove('active');
                    cartOverlay.classList.remove('active');
                } else {
                    cartOverlay.classList.add('active');
                    cartSidebar.classList.add('active');
                }

            window.updateCartUI = function() {
                const cepSection = document.querySelector('.cep-section');
                const cartFooter = document.querySelector('.cart-footer-inline');
                const cartTotalValue = document.getElementById('cartTotalValue');
                
                if (cart.length === 0) {
                    cartItemsContainer.innerHTML = '<div class="empty-cart-msg">Seu carrinho está vazio.</div>';
                    if (cartTotalValue) cartTotalValue.innerText = 'R$ 0,00';
                    cartCountBadge.innerText = '0';
                    if (cepSection) cepSection.classList.remove('show');
                    if (cartFooter) cartFooter.style.display = 'none';
                    return;
                }

                if (cepSection) cepSection.classList.add('show');
                let total = 0;
                let count = 0;
                cartItemsContainer.innerHTML = cart.map((item, index) => {
                    total += item.price * item.quantity;
                    count += item.quantity;
                    return `
                        <div class="cart-item" style="position: relative;">
                            <button class="remove-item" onclick="window.removeItem(${index})" title="Remover item">
                                <i class="fas fa-trash-alt"></i>
                            </button>
                            <div class="cart-item-img">
                                <img src="${item.image}" alt="${item.name}">
                            </div>
                            <div class="cart-item-info">
                                <span class="cart-item-name" style="padding-right: 20px;">${item.name}</span>
                                <span class="cart-item-variant">${item.variant}</span>
                                <span class="cart-item-price">R$ ${item.price.toLocaleString('pt-BR', { minimumFractionDigits: 2 })}</span>
                                <div class="cart-item-quantity-control">
                                    <button class="qty-btn" onclick="window.decreaseQty(${index})">−</button>
                                    <span class="qty-display">${item.quantity}</span>
                                    <button class="qty-btn" onclick="window.increaseQty(${index})">+</button>

                            </div>

                    // Rolagem automática removida conforme solicitação
                    
                    const shippingCost = selectedDelivery ? parseFloat(selectedDelivery.value) : 0;

                    if (cartFooter) cartFooter.style.display = 'none';

            function addItem() {
                const activeOption = document.querySelector('.option-card.active');
                const productName = document.getElementById('main-product-title').innerText;
                const variantName = activeOption.querySelector('.title').innerText;
                const priceText = activeOption.querySelector('.val').innerText.replace('R$ ', '').replace('.', '').replace(',', '.');
                const price = parseFloat(priceText);
                const image = variantImages[currentVariantIndex][0];

                const itemId = `${productName}-${variantName}`;
                const existingItem = cart.find(item => item.id === itemId);

                if (existingItem) {
                    existingItem.quantity += 1;
                } else {
                    cart.push({
                        id: itemId,
                        name: productName,
                        variant: variantName,
                        price: price,
                        image: image,
                        quantity: 1
                    });
                }

                window.updateCartUI();
                if (!cartSidebar.classList.contains('active')) toggleCart();
            }

            // Expose quantity control functions globally
            window.increaseQty = function(index) {
                if (index >= 0 && index < cart.length) {
                    cart[index].quantity += 1;
                    window.updateCartUI();
                }
            };

            window.decreaseQty = function(index) {
                if (index >= 0 && index < cart.length) {
                    if (cart[index].quantity > 1) {
                        cart[index].quantity -= 1;
                        window.updateCartUI();
                    } else {
                        window.removeItem(index);
                    }
                }
            };

            window.removeItem = function(index) {
                if (index >= 0 && index < cart.length) {
                    if (confirm('Deseja remover este item do carrinho?')) {
                        cart.splice(index, 1);
                        window.updateCartUI();
                    }
                }
            };

            // CEP Search Function
            window.searchCEP = async function() {
                const cepInput = document.getElementById('cepInput');
                const cep = cepInput.value.replace(/\D/g, '');
                const cepError = document.getElementById('cepError');
                const cepLoading = document.getElementById('cepLoading');
                const addressBlock = document.getElementById('addressBlock');
                const addressNumber = document.getElementById('addressNumber');
                const addressComplement = document.getElementById('addressComplement');
                const searchBtn = document.getElementById('searchCepBtn');

                cepError.classList.remove('show');
                cepError.textContent = '';

                if (cep.length !== 8) {
                    cepError.textContent = 'CEP deve conter 8 digitos';
                    cepError.classList.add('show');
                    return;
                }

                cepLoading.classList.add('show');
                searchBtn.disabled = true;
                searchBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i>';

                try {
                    // Artificial delay of 2 seconds
                    await new Promise(resolve => setTimeout(resolve, 2000));

                    const response = await fetch(`https://viacep.com.br/ws/${cep}/json/`);
                    const data = await response.json();

                    if (data.erro) {
                        searchBtn.innerHTML = 'Buscar';
                        cepError.textContent = 'CEP nao encontrado';
                        cepError.classList.add('show');
                        addressBlock.classList.remove('show');
                        addressNumber.style.display = 'none';
                        addressComplement.style.display = 'none';
                        document.getElementById('deliveryOptions').style.display = 'none';
                    } else {
                        document.getElementById('addressStreet').textContent = data.logradouro || 'N/A';
                        document.getElementById('addressNeighborhood').textContent = data.bairro || 'N/A';
                        document.getElementById('addressCity').textContent = data.localidade || 'N/A';
                        document.getElementById('addressState').textContent = data.uf || 'N/A';
                        addressBlock.classList.add('show');
                        addressNumber.style.display = 'block';
                        addressComplement.style.display = 'block';
                        document.getElementById('deliveryOptions').style.display = 'block';
                        addressNumber.value = '';
                        addressComplement.value = '';
                        
                        // Success state for button
                        searchBtn.style.background = 'var(--success)';
                        searchBtn.innerHTML = '<i class="fas fa-check"></i>';

                        // Atualiza UI para mostrar o botão de checkout
                        window.updateCartUI();
                    }
                } catch (error) {
                    searchBtn.innerHTML = 'Buscar';
                    cepError.textContent = 'Erro ao buscar CEP. Tente novamente';
                    cepError.classList.add('show');
                    addressBlock.classList.remove('show');
                    addressNumber.style.display = 'none';
                    addressComplement.style.display = 'none';
                    document.getElementById('deliveryOptions').style.display = 'none';
                } finally {
                    cepLoading.classList.remove('show');
                    searchBtn.disabled = false;
                }
            };

            window.handlePaymentRedirect = function() {
                const nameInput = document.getElementById('checkoutName');
                const cpfInput = document.getElementById('checkoutCPF');
                const phoneInput = document.getElementById('checkoutPhone');
                const emailInput = document.getElementById('checkoutEmail');
                const btn = event.currentTarget;

                // Reset error states
                [nameInput, cpfInput, phoneInput, emailInput].forEach(input => {
                    input.classList.remove('input-error');
                    input.addEventListener('input', () => input.classList.remove('input-error'), { once: true });
                });

                let firstError = null;

                if (!nameInput.value.trim()) {
                    nameInput.classList.add('input-error');
                    if (!firstError) firstError = nameInput;
                }
                if (!cpfInput.value.trim() || cpfInput.value.length < 14) {
                    cpfInput.classList.add('input-error');
                    if (!firstError) firstError = cpfInput;
                }
                if (!phoneInput.value.trim() || phoneInput.value.length < 14) {
                    phoneInput.classList.add('input-error');
                    if (!firstError) firstError = phoneInput;
                }
                if (!emailInput.value.trim() || !emailInput.value.includes('@')) {
                    emailInput.classList.add('input-error');
                    if (!firstError) firstError = emailInput;
                }

                if (firstError) {
                    firstError.focus();
                    // Smooth scroll to the field if needed
                    firstError.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    return;
                }

                // Show loading state
                btn.classList.add('loading');

                const subtotal = document.getElementById('checkoutSubtotal').innerText.replace('R$ ', '').replace('.', '').replace(',', '.');
                const address = document.getElementById('summaryAddress').innerText;
                const cep = document.getElementById('cepInput').value;
                const deliveryTime = document.getElementById('summaryShipping').innerText;
                
                const baseUrl = 'https://exemplo.com/';
                const params = new URLSearchParams({
                    subtotal: subtotal,
                    address: address,
                    cep: cep,
                    delivery_time: deliveryTime
                });
                
                // Small delay to show the loading effect before redirecting
                setTimeout(() => {
                    window.location.href = `${baseUrl}?${params.toString()}`;
                    
                    // Fallback: if the user stays on the page (e.g. pop-up blocked or slow redirect), 
                    // reset after a few seconds so the button isn't stuck forever
                    setTimeout(() => btn.classList.remove('loading'), 5000);
                }, 800);
            };
        })();
